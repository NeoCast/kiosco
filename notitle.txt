   Protected Sub cmd_Generar_Click(ByVal sender As Object, ByVal e As System.EventArgs) Handles cmd_Generar.Click
        Dim lo_input As New wsDocumentaciones.ent_input_documentacion

        Try
            io_ws.Url = txt_ws.Text

            'Cargar input
            lo_input.p_archivo_base64_c = txt_base64.Text
            lo_input.p_canal_c = txt_canal.Text
            lo_input.p_contrato_c = txt_contrato.Text
            lo_input.p_cuit_c = txt_cuit.Text
            lo_input.p_extension_c = txt_extension.Text
            lo_input.p_nombre_archivo_c = txt_nombre_archivo.Text
            lo_input.p_per_id_c = txt_cliente.Text
            lo_input.p_solicitud_i = txt_solicitud.Text
            lo_input.p_tipo_documento_c = txt_tipo_doc.Text
            lo_input.p_comentario_c = txt_comentario.Text
            lo_input.p_accion_c = txt_accion.Text

            io_ret = io_ws.f_recepcion_documentacion(lo_input)
            'Response.Write(String.Format("Codigo: {0} - Desc: {1} - id_archivo: {2}", io_ret.p_cod_error_i, io_ret.p_descripcion_error_c, io_ret.p_id_archivo_i))

            If io_ret.p_cod_error_i >= 0 Then
                Me.mensaje_exito.Visible = True
                Me.mensaje_error.Visible = False
                'Retorno
                txt_id_archivo_i.Text = io_ret.p_id_archivo_i
                txt_cod_error_i.Text = io_ret.p_cod_error_i
                txt_desc_error_c.Text = io_ret.p_descripcion_error_c
            Else
                Me.mensaje_exito.Visible = False
                Me.mensaje_error.Visible = True
                'Retorno               
                txt_cod_error_e.Text = io_ret.p_cod_error_i
                txt_desc_error_e.Text = io_ret.p_descripcion_error_c
            End If
        Catch ex As Exception
            Me.mensaje_exito.Visible = False
            Me.mensaje_error.Visible = True
            Me.txt_desc_error_e.Text = ex.Message
            'Response.Write(ex.Message)
        End Try
    End Sub
